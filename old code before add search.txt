def extract_numbers_from_text(text: str) -> str:
    """استخراج الأرقام من النص"""
    numbers = re.findall(r'\d+', text)
    return numbers[0] if numbers else ""

def normalize_medicine_name(name: str) -> str:
    """تطبيع اسم الدواء للبحث - الحفاظ على الأرقام"""
    if not name:
        return ""

    name = name.lower().strip()
    name = re.sub(r'O(?=\d)', '0', name)
    name = re.sub(r'(?<=\d)O', '0', name)
    name = name.replace('Omg', '40mg')
    name = name.replace('omg', '40mg')
    name = name.replace('Ome', '40me')
    name = name.replace('ome', '40me')

    return name

def exact_match(name: str) -> dict:
    """البحث عن تطابق تام"""
    if not name:
        return None

    name_lower = name.lower().strip()

    if name_lower in medicine_details:
        details = medicine_details[name_lower].copy()
        details["confidence"] = 100
        details["match_type"] = "exact"
        return details

    name_without_dose = re.sub(r'\d+mg', '', name_lower).strip()
    for med_name, details in medicine_details.items():
        med_without_dose = re.sub(r'\d+mg', '', med_name).strip()
        if name_without_dose == med_without_dose:
            result = details.copy()
            result["confidence"] = 95
            result["match_type"] = "exact_ignore_dose"
            return result

    return None

@lru_cache(maxsize=128)
def get_medicine_names_list():
    """إرجاع قائمة أسماء الأدوية للتخزين المؤقت"""
    return tuple(names)  # tuple لتكون hashable للتخزين المؤقت

def find_drug(name, threshold=0.65):
    """البحث عن دواء - محسن للسرعة"""
    if not name or len(name.strip()) < 2:
        return None

    name_clean = name.strip()
    cache_key = f"find_{name_clean}_{threshold}"

    if cache_key in search_cache:
        return search_cache[cache_key]

    # 1. تطابق تام
    exact_result = exact_match(name_clean)
    if exact_result:
        search_cache[cache_key] = exact_result["Name"]
        return exact_result["Name"]

    # 2. تطبيع الاسم
    name_normalized = normalize_medicine_name(name_clean)
    name_without_numbers = re.sub(r'\d+', '', name_normalized).strip()

    # 3. استخدام get_close_matches للبحث السريع
    names_tuple = get_medicine_names_list()

    # البحث بالاسم الطبيعي
    matches = difflib.get_close_matches(name_without_numbers, names_tuple, n=1, cutoff=threshold)

    if matches:
        result = matches[0]
        search_cache[cache_key] = result
        return result

    # 4. محاولة البحث مع تجاهل بعض الكلمات
    search_cache[cache_key] = None
    return None

def find_drug_with_details(name):
    """البحث عن دواء وإرجاع جميع التفاصيل - جميع الحقول"""
    if not name:
        return None

    cache_key = f"details_{name}"
    if cache_key in search_cache:
        return search_cache[cache_key]

    # 1. تطابق تام
    exact_result = exact_match(name)
    if exact_result:
        exact_result["alternatives"] = get_alternatives(exact_result["Name"], exclude_name=exact_result["Name"])
        search_cache[cache_key] = exact_result
        return exact_result

    # 2. بحث عادي
    drug_name = find_drug(name)

    if not drug_name:
        search_cache[cache_key] = None
        return None

    details = medicine_details.get(drug_name.lower())
    confidence = calculate_confidence(name, drug_name)

    if details:
        details_with_confidence = details.copy()
        details_with_confidence["confidence"] = confidence
        details_with_confidence["match_type"] = "fuzzy"
        details_with_confidence["alternatives"] = get_alternatives(drug_name, exclude_name=drug_name)
        search_cache[cache_key] = details_with_confidence
        return details_with_confidence
    else:
        result = {
            "Name": drug_name,
            "Contains": "",
            "ProductIntroduction": f"دواء {drug_name}",
            "ProductBenefits": "",
            "SideEffect": "",
            "HowToUse": "",
            "HowWorks": "",
            "QuickTips": "",
            "SafetyAdvice": "",
            "confidence": confidence,
            "match_type": "fuzzy",
            "alternatives": get_alternatives(drug_name, exclude_name=drug_name)
        }
        search_cache[cache_key] = result
        return result

def calculate_confidence(input_name: str, matched_name: str) -> int:
    """حساب نسبة الثقة"""
    input_clean = input_name.strip().lower()
    matched_clean = matched_name.strip().lower()

    if input_clean == matched_clean:
        return 100

    input_dose = re.findall(r'\d+', input_clean)
    matched_dose = re.findall(r'\d+', matched_clean)

    input_without_dose = re.sub(r'\d+', '', input_clean)
    matched_without_dose = re.sub(r'\d+', '', matched_clean)

    if input_without_dose == matched_without_dose:
        if input_dose and matched_dose and input_dose[0] == matched_dose[0]:
            return 95
        return 85

    ratio = difflib.SequenceMatcher(None, input_without_dose, matched_without_dose).ratio()
    return int(ratio * 100)

def get_alternatives(drug_name, exclude_name=None, limit=5):
    """الحصول على أدوية بديلة مشابهة"""
    alternatives = []
    drug_normalized = normalize_medicine_name(drug_name)
    drug_without_numbers = re.sub(r'\d+', '', drug_normalized).strip()

    for name in names:
        if exclude_name and name.lower() == exclude_name.lower():
            continue

        name_normalized = normalize_medicine_name(name)
        name_without_numbers = re.sub(r'\d+', '', name_normalized).strip()
        score = difflib.SequenceMatcher(None, drug_without_numbers, name_without_numbers).ratio()

        if score > 0.4:
            alternatives.append({"name": name, "score": round(score * 100, 1)})

    unique = []
    seen = set()
    for alt in alternatives:
        if alt["name"] not in seen:
            seen.add(alt["name"])
            unique.append(alt)

    unique.sort(key=lambda x: x["score"], reverse=True)
    return unique[:limit]

            FROM INFORMATION_SCHEMA.TABLES
            WHERE TABLE_TYPE = 'BASE TABLE'

        column_mapping = {

            'HowToUse': None,
            'HowWorks': None,

        }

        for expected in column_mapping.keys():
            for col in columns:
                if col.lower() == expected.lower():
                    column_mapping[expected] = col
                    break

        print(f"🔄 تطابق الأعمدة: {column_mapping}")

        # قراءة البيانات باستخدام fetchmany (أسرع وأكثر أماناً للبيانات الكبيرة)
        cursor.execute(f"SELECT * FROM [{table_name}]")

        count = 0
        error_count = 0
        batch_size = 5000  # قراءة 5000 صف في كل مرة

        print("⏳ جاري تحميل البيانات...")

        while True:
            rows = cursor.fetchmany(batch_size)
            if not rows:
                break

            for row in rows:
                try:
                    # استخراج الاسم
                    name_col = column_mapping['Name']
                    if name_col is None:
                        name_col = columns[0]

                    name_idx = columns.index(name_col)
                    name = str(row[name_idx]).strip() if row[name_idx] else ""

                    if name and name.lower() != 'nan' and name != '':
                        # دالة مساعدة لقراءة القيم بأمان
                        def safe_get(col_name):
                            if col_name and col_name in columns:
                                idx = columns.index(col_name)
                                val = row[idx]
                                if val is None:
                                    return ""
                                # تحويل أي نوع بيانات لـ string
                                if isinstance(val, (bytes, bytearray)):
                                    return val.decode('utf-8', errors='ignore').strip()
                                return str(val).strip()
                            return ""

                        medicine_details[name.lower()] = {
                            "Name": name,
                            "Contains": safe_get(column_mapping['Contains']),
                            "ProductIntroduction": safe_get(column_mapping['ProductIntroduction']),
                            "ProductBenefits": safe_get(column_mapping['ProductBenefits']),
                            "SideEffect": safe_get(column_mapping['SideEffect']),
                            "HowToUse": safe_get(column_mapping['HowToUse']),
                            "HowWorks": safe_get(column_mapping['HowWorks']),
                            "QuickTips": safe_get(column_mapping['QuickTips']),
                            "SafetyAdvice": safe_get(column_mapping['SafetyAdvice']),
                        }

        if error_count > 0:

        return False

# ==================== دوال البحث ====================

def extract_numbers_from_text(text: str) -> str:
    """استخراج الأرقام من النص"""
    numbers = re.findall(r'\d+', text)
    return numbers[0] if numbers else ""


def normalize_medicine_name(name: str) -> str:
    """تطبيع اسم الدواء للبحث - الحفاظ على الأرقام"""
    if not name:
        return ""

    name = name.lower().strip()
    name = re.sub(r'O(?=\d)', '0', name)
    name = re.sub(r'(?<=\d)O', '0', name)
    name = name.replace('Omg', '40mg')
    name = name.replace('omg', '40mg')
    name = name.replace('Ome', '40me')
    name = name.replace('ome', '40me')

    return name


def exact_match(name: str) -> Optional[Dict]:
    """البحث عن تطابق تام"""
    if not name:
        return None

    name_lower = name.lower().strip()

    if name_lower in medicine_details:
        details = medicine_details[name_lower].copy()
        details["confidence"] = 100
        details["match_type"] = "exact"
        return details

    name_without_dose = re.sub(r'\d+mg', '', name_lower).strip()
    for med_name, details in medicine_details.items():
        med_without_dose = re.sub(r'\d+mg', '', med_name).strip()
        if name_without_dose == med_without_dose:
            result = details.copy()
            result["confidence"] = 95
            result["match_type"] = "exact_ignore_dose"
            return result

    return None


@lru_cache(maxsize=128)
def get_medicine_names_list():
    """إرجاع قائمة أسماء الأدوية للتخزين المؤقت"""
    return tuple(all_medicine_names)


def find_drug(name, threshold=0.65):
    """البحث عن دواء - محسن للسرعة"""
    if not name or len(name.strip()) < 2:
        return None

    name_clean = name.strip()
    cache_key = f"find_{name_clean}_{threshold}"

    if cache_key in search_cache:
        return search_cache[cache_key]

    # 1. تطابق تام
    exact_result = exact_match(name_clean)
    if exact_result:
        search_cache[cache_key] = exact_result["Name"]
        return exact_result["Name"]

    # 2. تطبيع الاسم
    name_normalized = normalize_medicine_name(name_clean)
    name_without_numbers = re.sub(r'\d+', '', name_normalized).strip()

    # 3. استخدام get_close_matches للبحث السريع
    names_tuple = get_medicine_names_list()

    # البحث بالاسم الطبيعي
    matches = difflib.get_close_matches(name_without_numbers, names_tuple, n=1, cutoff=threshold)

    if matches:
        result = matches[0]
        search_cache[cache_key] = result
        return result

    # 4. محاولة البحث مع تجاهل بعض الكلمات
    search_cache[cache_key] = None
    return None


def calculate_confidence(input_name: str, matched_name: str) -> int:
    """حساب نسبة الثقة"""
    input_clean = input_name.strip().lower()
    matched_clean = matched_name.strip().lower()

    if input_clean == matched_clean:
        return 100

    input_dose = re.findall(r'\d+', input_clean)
    matched_dose = re.findall(r'\d+', matched_clean)

    input_without_dose = re.sub(r'\d+', '', input_clean)
    matched_without_dose = re.sub(r'\d+', '', matched_clean)

    if input_without_dose == matched_without_dose:
        if input_dose and matched_dose and input_dose[0] == matched_dose[0]:
            return 95
        return 85

    ratio = difflib.SequenceMatcher(None, input_without_dose, matched_without_dose).ratio()
    return int(ratio * 100)


def get_alternatives(drug_name, exclude_name=None, limit=5):
    """الحصول على أدوية بديلة مشابهة"""
    alternatives = []
    drug_normalized = normalize_medicine_name(drug_name)
    drug_without_numbers = re.sub(r'\d+', '', drug_normalized).strip()

    for name in all_medicine_names:
        if exclude_name and name.lower() == exclude_name.lower():
            continue

        name_normalized = normalize_medicine_name(name)
        name_without_numbers = re.sub(r'\d+', '', name_normalized).strip()
        score = difflib.SequenceMatcher(None, drug_without_numbers, name_without_numbers).ratio()

        if score > 0.4:
            alternatives.append({"name": name, "score": round(score * 100, 1)})

    unique = []
    seen = set()
    for alt in alternatives:
        if alt["name"] not in seen:
            seen.add(alt["name"])
            unique.append(alt)

    unique.sort(key=lambda x: x["score"], reverse=True)
    return unique[:limit]


def find_drug_with_details(name):
    """البحث عن دواء وإرجاع جميع التفاصيل - جميع الحقول"""
    if not name:
        return None

    cache_key = f"details_{name}"
    if cache_key in search_cache:
        return search_cache[cache_key]

    # 1. تطابق تام
    exact_result = exact_match(name)
    if exact_result:
        exact_result["alternatives"] = get_alternatives(exact_result["Name"], exclude_name=exact_result["Name"])
        search_cache[cache_key] = exact_result
        return exact_result

    # 2. بحث عادي
    drug_name = find_drug(name)

    if not drug_name:
        search_cache[cache_key] = None
        return None

    details = medicine_details.get(drug_name.lower())
    confidence = calculate_confidence(name, drug_name)

    if details:
        details_with_confidence = details.copy()
        details_with_confidence["confidence"] = confidence
        details_with_confidence["match_type"] = "fuzzy"
        details_with_confidence["alternatives"] = get_alternatives(drug_name, exclude_name=drug_name)
        search_cache[cache_key] = details_with_confidence
        return details_with_confidence
    else:
        result = {
            "Name": drug_name,
            "Contains": "",
            "ProductIntroduction": f"دواء {drug_name}",
            "ProductBenefits": "",
            "SideEffect": "",
            "HowToUse": "",
            "HowWorks": "",
            "QuickTips": "",
            "SafetyAdvice": "",
            "confidence": confidence,
            "match_type": "fuzzy",
            "alternatives": get_alternatives(drug_name, exclude_name=drug_name)
        }
        search_cache[cache_key] = result
        return result


def get_all_medicine_names():
    return all_medicine_names


def search_medicine_by_name(name):
    return find_drug_with_details(name)


def get_suggestions(partial_name: str, limit: int = 10) -> List[Dict]:
    """الحصول على قائمة اقتراحات بناءً على نص جزئي - لـ autocomplete مثل Google"""
    if not partial_name or len(partial_name.strip()) < 1:
        return []

    partial_clean = partial_name.strip().lower()
    names_tuple = get_medicine_names_list()

    # المطابقات التي تبدأ بنفس النص
    prefix_matches = []
    exact_prefix_matches = []

    for name in names_tuple:
        name_lower = name.lower()
        if name_lower.startswith(partial_clean):
            if name_lower == partial_clean:
                exact_prefix_matches.append({"name": name, "type": "exact", "score": 100})
            else:
                prefix_matches.append({"name": name, "type": "prefix", "score": 95})

    # البحث عن مطابقات مشابهة باستخدام difflib
    fuzzy_matches = difflib.get_close_matches(partial_clean, [n.lower() for n in names_tuple], n=limit, cutoff=0.3)
    fuzzy_results = []
    for match in fuzzy_matches:
        original_name = next((n for n in names_tuple if n.lower() == match), match)
        if original_name:
            fuzzy_results.append({"name": original_name, "type": "fuzzy", "score": 80})

    # دمج النتائج مع إزالة التكرار
    suggestions = []
    seen = set()

    # الأولوية: المطابقات التامة
    for item in exact_prefix_matches:
        if item["name"] not in seen:
            suggestions.append(item)
            seen.add(item["name"])

    # ثم المطابقات البادئة
    for item in prefix_matches:
        if item["name"] not in seen:
            suggestions.append(item)
            seen.add(item["name"])

    # ثم المطابقات المشابهة
    for item in fuzzy_results:
        if item["name"] not in seen:
            suggestions.append(item)
            seen.add(item["name"])

    return suggestions[:limit]


def clear_cache():
    """مسح ذاكرة التخزين المؤقت"""
    global search_cache
    search_cache = {}
    get_medicine_names_list.cache_clear()
